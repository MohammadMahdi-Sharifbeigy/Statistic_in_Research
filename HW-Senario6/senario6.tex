	\section{مثال: مولفه N400 در پردازش معنایی جمله}
	
	\subsection{زمینه پژوهش}
	مطالعه بررسی مولفه N400 در پاسخ به جملات با پایان‌های معنایی منطبق (congruent) و ناسازگار (incongruent) انجام شده است. این مطالعه از طرح درون‌آزمودنی استفاده می‌کند که در آن همه شرکت‌کنندگان در تمام شرایط آزمایشی شرکت می‌کنند.
	
	\subsection{سوال پژوهشی}
	آیا دامنه مولفه N400 در سه شرایط مختلف جمله (منطبق، ناسازگار، بی‌معنا) تفاوت معناداری دارد؟
	
	\subsection{فرضیه‌ها}
	
	\textbf{فرضیه صفر ($H_0$):} میانگین دامنه N400 در همه شرایط برابر است.\\
	\textbf{فرضیه جایگزین ($H_1$):} میانگین دامنه N400 در حداقل یکی از شرایط متفاوت است.
	
	\subsection{ساختار داده‌ها}
	
	\textbf{عامل درون‌آزمودنی:} شرایط جمله (سه سطح: منطبق، ناسازگار، بی‌معنا). هر شرکت‌کننده همه شرایط را تجربه می‌کند.\\
	\textbf{متغیر وابسته:} دامنه میانگین N400 (میکروولت) در پنجره زمانی 300-500 میلی‌ثانیه.

	\subsubsection{مفهوم مولفه N400}
	
	\textbf{N400 چیست؟}
	\begin{itemize}
		\item پتانسیل برانگیخته منفی با اوج حدود 400 میلی‌ثانیه
		\item شاخص پردازش معنایی و یکپارچگی معنایی
		\item حداکثر در نواحی مرکزی-آهیانه‌ای (\lr{Cz, Pz})
		\item حساس به انتظارات معنایی و احتمال بسته‌شدن (\lr{Cloze Probability})
	\end{itemize}
	
	\textbf{مولفه‌های همراه:}
	\begin{itemize}
		\item \textbf{پتانسیل تشخیص (\lr{RP}):} حدود 250 میلی‌ثانیه، مربوط به تشخیص کلمه
		\item \textbf{مولفه مثبت دیرهنگام (\lr{LPC/P600}):} حدود 600 میلی‌ثانیه، مرتبط با بازتحلیل معنایی
		\item \textbf{توزیع فضایی:} N400 عمدتاً در نیمکره چپ و نواحی گیجگاهی
	\end{itemize}

	\subsubsection{تحلیل اصلی}
	
	\textbf{آزمون t تک‌نمونه‌ای:}
	در الکترود Cz در پنجره 395-405 میلی‌ثانیه:
	$$t = \frac{\bar{X}_{incongruent} - \bar{X}_{congruent}}{SE_{difference}}$$
	
	\textbf{تحلیل واریانس اندازه‌گیری‌های مکرر:}
	$$F = \frac{MS_{between}}{MS_{error}}$$
	
	با درجات آزادی: $df_{between} = k-1$، $df_{error} = (n-1)(k-1)$
	
	که $k$ تعداد شرایط و $n$ تعداد آزمودنی‌ها است.
	
	\subsubsection{تحلیل‌های تکمیلی}
	
	\textbf{عوامل مورد بررسی:}
	\begin{itemize}
		\item احتمال بسته‌شدن (\lr{Close Probability})
		\item ترتیب ارائه (منطبق یا ناسازگار اول)
		\item تقسیم‌بندی زبانی (خطای معنایی فقط یا معنایی+دستوری)
		\item ربع آزمایش (اثر خستگی)
	\end{itemize}
	
	\textbf{تصحیح چندگانه:}
	\begin{itemize}
		\item Bonferroni برای آزمون‌های t چندگانه
		\item تصحیح کرویت با Greenhouse-Geisser در صورت نیاز
	\end{itemize}

	\section{تفسیر نتایج}
	
	\subsection{یافته‌های اصلی}
	
	\textbf{رد فرضیه صفر:}
	فرضیه صفر در سطح $p < 0.001$ رد شد. جملات ناسازگار N400 بزرگ‌تر و معناداری نسبت به جملات منطبق ایجاد کردند.
	
	\textbf{الگوی نتایج:}
	\begin{itemize}
		\item \textbf{سلسله‌مراتب N400:} بی‌معنا > ناسازگار > منطبق
		\item \textbf{توزیع فضایی:} مرکزی-آهیانه‌ای با گسترش به نواحی گیجگاهی چپ
		\item \textbf{مولفه‌های همراه:} RP در شرایط منطبق، LPC در شرایط ناسازگار
	\end{itemize}
	
	\subsection{تفسیر علمی}
	
	\textbf{N400 به عنوان شاخص یکپارچگی معنایی:}
	\begin{itemize}
		\item دامنه بزرگ‌تر N400 نشان‌دهنده دشواری بیشتر در یکپارچه‌سازی معنایی
		\item جملات ناسازگار نیازمند تلاش بیشتر برای پردازش معنایی
		\item الگوی سلسله‌مراتبی تأیید نظریه پیش‌بینی معنایی
	\end{itemize}
	
	\textbf{اهمیت مولفه‌های همراه:}
	\begin{itemize}
		\item \lr{RP:} تشخیص زودهنگام انطباق معنایی
		\item \lr{LPC/P600}: بازتحلیل و بازارزیابی معنایی
		\item توزیع فضایی متفاوت نشان‌دهنده شبکه‌های عصبی مجزا
	\end{itemize}
	
	\section{آزمون‌های تعقیبی}
	
	\subsection{مقایسات چندگانه}
	
	پس از یافتن تفاوت معنادار در ANOVA، آزمون‌های تعقیبی Bonferroni انجام شد:
	
	\textbf{مقایسه‌های جفتی در N400 \lr{(400 ms)}:}
	\begin{itemize}
		\item ناسازگار vs منطبق: $p < 0.001$، $d = 0.99$
		\item بی‌معنا vs منطبق: $p < 0.001$، $d = 1.24$
		\item بی‌معنا vs ناسازگار: $p = 0.003$، $d = 0.68$
	\end{itemize}
	
	\section{تحلیل توپوگرافی}
	
	\subsection{توزیع فضایی N400}
	
	\textbf{تحلیل خوشه‌ای فضایی-زمانی:}
	\begin{itemize}
		\item خوشه‌های منفی معنادار در نواحی مرکزی-آهیانه‌ای (\lr{250-600 ms})
		\item خوشه‌های مثبت در نواحی پیشانی-گیجگاهی چپ
		\item گسترش به نواحی آهیانه‌ای-پس‌سری در زمان‌های دیرتر
	\end{itemize}
	
	\textbf{الگوی زمان-فضا:}
	\begin{itemize}
		\item \lr{150-300 ms}: فعالیت پیشانی چپ (RP)
		\item \lr{250-600 ms}: فعالیت مرکزی-آهیانه‌ای (\lr{N400})
		\item \lr{600-850 ms}: فعالیت گیجگاهی-آهیانه‌ای (LPC)
	\end{itemize}
	
	\section{محدودیت‌ها و ملاحظات}
	
	\subsection{محدودیت‌های روش‌شناختی}
	
	\begin{itemize}
		\item \textbf{ترتیب ثابت:} محرک‌ها به ترتیب یکسان ارائه شدند
		\item \textbf{اثر تکرار:} هر جمله دو بار (منطبق و ناسازگار) ارائه شد
		\item \textbf{فقدان آهنگ کلامی:} ممکن است طبیعی نباشد
		\item \textbf{حجم نمونه:} 20 نفر برای تعمیم‌پذیری محدود
	\end{itemize}
	
	\subsection{فرضیات \lr{Repeated Measures ANOVA}}
	
	\textbf{فرضیات برقرار شده:}
	\begin{itemize}
		\item متغیر وابسته پیوسته: دامنه N400
		\item نمونه‌های وابسته: همه آزمودنی‌ها همه شرایط را تجربه کردند
		\item نرمال بودن: تقریباً برقرار در همه شرایط
	\end{itemize}
	
	\textbf{نقض کرویت (Sphericity):}
	\begin{itemize}
		\item آزمون Mauchly معنادار در برخی مقایسه‌ها
		\item تصحیح Greenhouse-Geisser اعمال شد
		\item نتایج پس از تصحیح همچنان معنادار باقی ماندند
	\end{itemize}

	\section{فرمول‌های آماری}
	
	\subsection{تحلیل واریانس اندازه‌گیری‌های مکرر}
	
	\begin{equation}
		F = \frac{MS_{between}}{MS_{error}}
	\end{equation}
	
	که در آن:
	\begin{itemize}
		\item $MS_{between} = \frac{SS_{between}}{df_{between}}$
		\item $MS_{error} = \frac{SS_{error}}{df_{error}}$
		\item $df_{between} = k - 1$ (\lr{k} = تعداد شرایط)
		\item $df_{error} = (n-1)(k-1)$ (\lr{n} = تعداد آزمودنی‌ها)
	\end{itemize}
	
	\subsection{تصحیح کرویت}
	
	درصورت نقض فرضیه کرویت:
	
	\begin{equation}
		F_{corrected} = \frac{MS_{between}}{MS_{error}} \times \varepsilon
	\end{equation}
	
	که $\varepsilon$ ضریب تصحیح Greenhouse-Geisser است.
	
	\subsection{اندازه اثر}
	
	\begin{equation}
		\eta^2_{partial} = \frac{SS_{effect}}{SS_{effect} + SS_{error}}
	\end{equation}
	
	راهنمای تفسیر:
	\begin{itemize}
		\item کوچک: $\eta^2 = 0.01$
		\item متوسط: $\eta^2 = 0.06$
		\item بزرگ: $\eta^2 = 0.14$
	\end{itemize}
	
	\subsection{تصحیح Bonferroni}
	
	برای $k$ مقایسه:
	
	\begin{equation}
		\alpha_{corrected} = \frac{\alpha}{k}
	\end{equation}
	
	یا محاسبه p-value تصحیح شده:
	
	\begin{equation}
		p_{corrected} = p_{observed} \times k
	\end{equation}
